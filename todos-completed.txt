Sample opnsense code:
  type pfStatesResponse struct {
      Current string `json:"current"`
      Limit   string `json:"limit"`
  }
  type PFStates struct {
      Current int
      Limit   int
  }
  func (c *Client) FetchPFStates() (PFStates, *APICallError) {
      var resp pfStatesResponse
      url, ok := c.endpoints["pfStates"]
      if !ok {
          return PFStates{}, &APICallError{Endpoint: "pfStates", Message: "endpoint not found"}
      }
      if err := c.do("GET", url, nil, &resp); err != nil {
          return PFStates{}, err
      }
      current, _ := parseStringToInt(resp.Current, url)
      limit, _ := parseStringToInt(resp.Limit, url)
      return PFStates{Current: current, Limit: limit}, nil
  }

Note: parseStringToInt is already available in opnsense/utils.go.
